    ```javascript
    function loadPersonasConfig() {
      const configPath = path.resolve(__dirname, './config/personas.json');
      try {
        const raw = fs.readFileSync(configPath, 'utf-8');
        const config = JSON.parse(raw);
        const personas = config.personas || {};
        const defaultPersona = config.default || 'warm-senior';
        // Validate each persona has required fields
        for (const [id, p] of Object.entries(personas)) {
          if (!p.id || !p.name || !p.prompt) {
            console.warn(`Persona "${id}" is missing required fields, skipping`);
            delete personas[id];
          }
        }
        if (Object.keys(personas).length === 0) {
          throw new Error('No valid personas in config');
        }
        console.log(`Loaded ${Object.keys(personas).length} persona(s), default: ${defaultPersona}`);
        return { personas, defaultPersona };
      } catch (error) {
        console.warn('Failed to load personas.json, using built-in warm-senior only:', error.message);
        return {
          personas: {
            'warm-senior': { id: 'warm-senior', name: '温暖学长', prompt: '你的语气应该是温暖、支持性的，像一位耐心的学长。' }
          },
          defaultPersona: 'warm-senior'
        };
      }
    }
    ```

    B2. After `loadPersonasConfig`, add a function `loadPostsIndex()`:

    ```javascript
    function loadPostsIndex() {
      const indexPath = process.env.POSTS_INDEX_PATH
        ? (path.isAbsolute(process.env.POSTS_INDEX_PATH) ? process.env.POSTS_INDEX_PATH : path.resolve(__dirname, process.env.POSTS_INDEX_PATH))
        : path.resolve(__dirname, './data/posts-index.json');
      try {
        const raw = fs.readFileSync(indexPath, 'utf-8');
        const posts = JSON.parse(raw);
        if (!Array.isArray(posts) || posts.length === 0) {
          console.warn('posts-index.json is empty or invalid, content awareness disabled');
          return '';
        }
        // Format as compact text block for system prompt KNOWLEDGE section
        // Each post: title, date, tags, link — compact format to stay under 2500 tokens
        const lines = posts.map(p => {
          const tags = (p.tags || []).join(', ');
          const date = (p.date || '').slice(0, 10); // YYYY-MM-DD
          return `- ${p.title} (${date}) [${tags}] ${p.link}`;
        });
        const block = lines.join('\n');
        console.log(`Loaded ${posts.length} posts into index (${lines.length} lines)`);
        return block;
      } catch (error) {
        console.warn('posts-index.json not found or unreadable, content awareness disabled');
        return '';
      }
    }
    ```

    B3. Near the top-level constants (after `const systemPrompt = loadSystemPrompt();` on line 33), REPLACE that line with:
    ```javascript
    const personasConfig = loadPersonasConfig();
    const postsIndexText = loadPostsIndex();

    // Token budget monitoring — logs warning when composed prompt exceeds 80% of 2500 token target
    function estimateTokens(text) {
      // Rough approximation: 1 token ≈ 4 characters for Chinese, 1 token ≈ 4 chars for English
      // Conservative: use 3 chars/token for mixed CN/EN content
      return Math.ceil(text.length / 3);
    }

    function composeSystemPrompt(personaId) {
      const resolvedPersonaId = personaId && personasConfig.personas[personaId]
        ? personaId
        : (personasConfig.personas[personasConfig.defaultPersona] ? personasConfig.defaultPersona : 'warm-senior');

      if (personaId && !personasConfig.personas[personaId]) {
        console.warn(`Unknown persona_id "${personaId}", defaulting to "${resolvedPersonaId}"`);
      }

      const persona = personasConfig.personas[resolvedPersonaId];
      // PERSONA_IDENTITY note: loadSystemPrompt() still provides the fallback via config/system_prompt.txt or env var
      const baseIdentity = loadSystemPrompt();

      // Build layered prompt: ## IDENTITY → ## TONE → ## KNOWLEDGE
      // Bookend strategy: personality reminder at both start and end of TONE section
      const toneSection = `\n## TONE — 语气和风格\n${persona.prompt}\n\n请始终遵循以上语气风格。记住你的性格设定，在每一次回复中都要自然地体现出来。`;

      let knowledgeSection = '';
      if (postsIndexText) {
        knowledgeSection = `\n## KNOWLEDGE — 博客内容\n以下是博客中已发布文章的信息，你可以根据这些信息回答用户关于博客内容的问题，推荐相关文章，或讨论文章主题：\n\n${postsIndexText}\n\n当用户询问博客相关内容时，请基于以上真实文章信息回答。如果没有匹配的文章，诚实告知而不是编造不存在的内容。`;
      }

      const composed = `${baseIdentity}${toneSection}${knowledgeSection}`;

      // Token budget logging
      const estimatedTokens = estimateTokens(composed);
      const tokenBudget = 2500;
      if (estimatedTokens > tokenBudget * 0.8) {
        console.warn(`[TOKEN-BUDGET] System prompt estimated at ${estimatedTokens} tokens (${Math.round(estimatedTokens / tokenBudget * 100)}% of ${tokenBudget} target). persona_id="${resolvedPersonaId}", posts_index=${postsIndexText ? 'loaded' : 'empty'}`);
      }

      return composed;
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client POST body → server | `persona_id` is user-controlled string — must be validated against known presets |
| `POSTS_INDEX_PATH` env var → `fs.readFileSync` | Path from environment must be validated (no path traversal) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | composeSystemPrompt personaId lookup | mitigate | Reject unknown personaId values — only accept keys that exist in `personasConfig.personas` Map; default to "warm-senior" on mismatch (log warning) |
| T-01-02 | Tampering | personas.json on disk | accept | File is on server filesystem, not user-writable. Persona prompts are curated; content injection via personaId is not a vector since only known keys are accepted. |
| T-01-03 | Information Disclosure | TOKEN-BUDGET log | accept | Token estimates are approximate, logged to server console only, not exposed to client. No sensitive data in token counts. |
| T-01-04 | Denial of Service | POSTS_INDEX_PATH path traversal | mitigate | If POSTS_INDEX_PATH contains `..` or is absolute outside expected dir, `fs.readFileSync` is wrapped in try/catch — graceful fallback to empty posts index with log warning. |
</threat_model>
